import torch
import pickle
import seaborn as sn
import matplotlib.pyplot as plt
from model import ResNet_pt
from sklearn.metrics import confusion_matrix, matthews_corrcoef
import pandas as pd
device = torch.device("cpu")

model = ResNet_pt().to(device)
model.load_state_dict(torch.load('output/k_cross_ResNet_balanced_state_dict.pt',map_location=device))
model.to(device)
model.eval()

fig, axes = plt.subplots(2, 3, figsize=(15, 10))

for i, ax in enumerate(axes.flatten()):
    if i == 5:
        break
    test_loader = pickle.load(open(f'output/test_loader_fold{i}.pkl', 'rb'))
    y_true, y_pred = [], []
    with torch.no_grad():
        for data in test_loader:
            images, labels = data
            images, labels = images.to(device), labels.to(device)
            output = model(images)
            _, predictions = torch.max(output.data, 1)
            y_true.extend(labels.tolist())
            y_pred.extend(predictions.tolist())
    cm = confusion_matrix(y_true, y_pred)
    df_cm = pd.DataFrame(cm, index=['spoof', 'live'], columns=['spoof', 'live'])
    
    # Plot confusion matrix on the current subplot
    sn.heatmap(df_cm, annot=True, ax=ax)
    ax.set_title(f"Confusion matrix for fold {i+1}")

plt.tight_layout()
plt.show()
